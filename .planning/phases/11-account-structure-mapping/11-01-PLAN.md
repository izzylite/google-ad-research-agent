    ```python
        """Extract the `**Geo focus:**` value list from a brief markdown.

        Examples:
            >>> _parse_optional_geo_focus("- **Geo focus:** Palm Beach County, Lake Worth")

        if not raw:
    ```

       ```python
           """Load us-cities.json. Returns {state_code_lower: {city_lower: county_lower}}.

           Empty dict if file absent or unparseable (allows backward compat / test injection).
           """

           except (FileNotFoundError, json.JSONDecodeError):

       def _strip_county_suffix(name: str) -> str:
           """'Palm Beach County' → 'palm beach'. Used to normalize geo_focus entries
           before lookup against us-cities county values (which are stored without ' county').
           """
           return _COUNTY_SUFFIX_RE.sub("", name.strip()).lower()

       def _build_city_filter(
           state_code: str,
           geo_focus: list[str],
           us_cities: dict[str, dict[str, str]],
       ) -> dict[str, set[str]]:
           """Return {'in': <cities in focus>, 'out': <cities not in focus>} for state.

           City is 'in focus' if the city name itself or its county appears in geo_focus
           (case-insensitive; ' county' suffix stripped from geo_focus entries — Pitfall 5).
           Empty geo_focus → empty 'out' set (filter disabled, backward compat).
           """
           state_cities = us_cities.get(state_code.lower(), {})
           if not geo_focus or not state_cities:
               return {"in": set(), "out": set()}
           focus_norm = {_strip_county_suffix(g) for g in geo_focus}
           in_focus: set[str] = set()
           out_of_focus: set[str] = set()
           for city, county in state_cities.items():
               if city.lower() in focus_norm or county.lower() in focus_norm:
                   in_focus.add(city.lower())
               else:
                   out_of_focus.add(city.lower())
           return {"in": in_focus, "out": out_of_focus}

       def _keyword_drifts_city(text: str, city_filter: dict[str, set[str]]) -> bool:
           """True if keyword contains any out-of-focus city name (multi-word substring match).

           Empty city_filter['out'] → False (filter inactive, backward compat).
           Case-insensitive. Uses literal substring match because city names are
           multi-word (e.g., 'west palm beach', 'boca raton') and regex word boundaries
           would mis-handle these.
           """
           out = city_filter.get("out") or set()
           if not out:
               return False
           lower = text.lower()
           return any(c in lower for c in out)
       ```
