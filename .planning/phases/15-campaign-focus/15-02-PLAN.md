    ```python
    def render_campaign_focus_section(
        brief_fields: dict[str, str],
        *,
        perf_path: Path | None = None,
    ) -> str:
        """CAMP-05: render the '## Campaign Focus' callout + validate vs perf.json.

        Returns markdown with one of:
            - Empty string when brief_fields["campaign_focus"] is empty.
            - Single-campaign block: `**Campaign:** <name>`
            - Multi-campaign bulleted list.
            - Either form may include a `⚠ Campaign name not found in
              account: '<name>' — check for typo` line per mismatched name
              when perf_path is provided and the name is absent from the
              campaigns list.

        Pipes / angle brackets / smart quotes inside campaign names are NOT
        escaped via escape_md_cell — Google Ads campaign names like
        `Search | Lake Worth Accident Exams | Manual CPC` deliberately use
        pipes as a labelling convention, and escaping breaks operator
        recognition. (Compare with Phase 11 GEO-05 which DOES route through
        escape_md_cell because city names rarely contain markdown-special chars.)
        """
        raw = (brief_fields or {}).get("campaign_focus", "").strip()
        if not raw:
            return ""

        # Split list form: pipe-no-space ('A|B|C') is a list; space-pipe-space
        # (' | ') is part of a single campaign-naming convention. Mirrors the
        # heuristic in perf_fetch.py main_with_args.
        if "|" in raw and " | " not in raw:
            names = [n.strip() for n in raw.split("|") if n.strip()]
        else:
            names = [raw]

        # Validate against perf.json campaigns list when path provided.
        warnings: list[str] = []
        if perf_path is not None:
            try:
                perf_data = json.loads(perf_path.read_text(encoding="utf-8"))
                known = {c.get("name", "") for c in perf_data.get("campaigns", [])}
                for n in names:
                    if n not in known:
                        warnings.append(
                            f"> ⚠ Campaign name not found in account: '{n}' "
                            f"— check for typo\n"
                        )
            except (FileNotFoundError, json.JSONDecodeError, OSError):
                # Graceful: validation needs the file to exist + parse.
                pass

        parts = ["## Campaign Focus\n\n"]
        if len(names) == 1:
            parts.append(f"**Campaign:** {names[0]}\n\n")
        else:
            for n in names:
                parts.append(f"- {n}\n")
            parts.append("\n")
        parts.extend(warnings)
        return "".join(parts)
    ```
