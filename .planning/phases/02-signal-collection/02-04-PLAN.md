# merge_signals.py CLI contract:
#   uv run merge_signals.py --run-dir <abs>
#   stdout: {"keywords_count": N, "source_diversity_avg": float, "variants_merged": N}
#   exit 0 ok / exit 3 fatal (IO, run_dir not found)
#
# Source taxonomy (locked — all 6 must be handled):
#   "serper-organic", "serper-paa", "serper-related", "serper-ads"  (from serper.json by_seed[])
#   "tavily-extract"  (from tavily-<domain>.json results[])
#   "websearch-baseline"  (from websearch-baseline.json extracted_keywords[])
#
# keywords.json output row shape (from RESEARCH.md § Source Attribution Data Model):
#   {"canonical", "lemma_hash", "variants": [...], "signal_count", "source_diversity", "sources": [...]}
#   where sources[i] = {"source", "snippet"?, "url"?, "from_seed"?, "competitor_domain"?, "from_query"?, "captured_at"?}
#
# Merge algorithm:
#   1. Read all raw/*.json — handle serper.json (by_seed[].organic/paa/related/ads), tavily-*.json (results[]), websearch-baseline.json (extracted_keywords[])
#   2. For each keyword-bearing item, extract the keyword text and call canonicalise()
#   3. Group rows by lemma_hash
#   4. Per group: canonical = shortest surface form; variants = sorted list of all observed forms; sources = all source attribution dicts; signal_count = len(sources); source_diversity = len({s["source"] for s in sources})
#   5. Drop keywords longer than 7 words (Pitfall 6 filter)
#   6. Write keywords.json to run_dir root
</context>
