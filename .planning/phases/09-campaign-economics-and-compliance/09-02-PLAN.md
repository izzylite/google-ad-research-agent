CRITICAL — Pitfall 8 (cpc_micros unit conversion):
- Inputs (cpc_micros, suggested_max_cpc_micros) are in micros (USD × 1,000,000).
- All arithmetic stays in micros UNTIL the final USD field. Convert ONCE at the display boundary:
  `usd = micros / 1_000_000`
- Sanity: suggested_max_cpc_micros=320000 × 0.65 = 208000 micros avg CPC = $0.208 avg CPC.
  daily_clicks_mid=25 × $0.208 = $5.20 daily spend (NOT $5.2M).

CRITICAL — Pitfall 6 (cluster join):
- Build `ranked_index = {row["keyword"].lower().strip(): row for row in ranked_enriched}`.
- For each cluster keyword: lookup is `ranked_index.get(cluster_kw["keyword"].lower().strip())`.
- Track `unjoined_keywords` count in metadata — surface silent-failure signal.

Pitfall 5 (no over-counting at campaign level):
- campaign_totals.daily_clicks_mid = SUM of per-cluster daily_clicks_mid.
- DO NOT recompute from scratch; aggregate the cluster-level numbers.
- This keeps "skipped keywords" (no volume or no CPC) consistent between cluster and campaign rollups.

Per-cluster computation (FRCS-01..03):
```
for keyword in cluster.keywords:
    row = ranked_index.get(keyword.lower().strip())
    if row is None: unjoined += 1; continue
    volume = row.get("volume")
    sugg_cpc = row.get("suggested_max_cpc_micros")
    intent = row.get("intent")
    if volume is None or sugg_cpc is None: skipped += 1; continue
    ctr = INTENT_CTRS.get(intent, 0.0)
    monthly_clicks_kw = volume * ctr
    avg_cpc_micros = sugg_cpc * AVG_CPC_RATIO
    daily_clicks_kw = monthly_clicks_kw / 30.0
    daily_spend_micros_kw = daily_clicks_kw * avg_cpc_micros
    total_daily_clicks_mid += daily_clicks_kw
    total_daily_spend_micros_mid += daily_spend_micros_kw

# Apply bands at the cluster aggregate (FRCS-03):
daily_clicks_low  = int(round(total_daily_clicks_mid * 0.5))
daily_clicks_mid  = int(round(total_daily_clicks_mid * 1.0))
daily_clicks_high = int(round(total_daily_clicks_mid * 1.5))
daily_spend_mid_usd = round(total_daily_spend_micros_mid / 1_000_000, 2)
daily_spend_low_usd = round(daily_spend_mid_usd * 0.5, 2)
daily_spend_high_usd = round(daily_spend_mid_usd * 1.5, 2)
monthly_spend_mid_usd = round(daily_spend_mid_usd * 30, 2)
```

```python
def main_with_args(argv: list[str]) -> int:
    parser = argparse.ArgumentParser(prog="forecast_budget")
    parser.add_argument("--run-dir", required=True, type=Path)
    args_only = argv[1:] if argv and not argv[0].startswith("-") else argv
    args = parser.parse_args(args_only)
    
    ranked_path = args.run_dir / "ranked-enriched.json"
    clusters_path = args.run_dir / "clusters.json"
    
    for p, label in [(ranked_path, "ranked-enriched.json"), (clusters_path, "clusters.json")]:
        if not p.exists():
            log.error("%s not found at %s", label, p)
            return 3
    
    try:
        ranked = json.loads(ranked_path.read_text(encoding="utf-8"))
        clusters_data = json.loads(clusters_path.read_text(encoding="utf-8"))
    except json.JSONDecodeError as e:
        log.error("Failed to parse input JSON: %s", e)
        return 3
    
    forecast = build_forecast(ranked, clusters_data, run_id=args.run_dir.name)
    
    out = args.run_dir / "forecast.json"
    tmp = out.with_suffix(".json.tmp")
    tmp.write_text(json.dumps(forecast, indent=2, ensure_ascii=False), encoding="utf-8")
    tmp.replace(out)
    
    summary = {
        "clusters_forecast": len(forecast["clusters"]),
        "keywords_in_forecast": forecast["campaign_totals"]["keyword_count"],
        "daily_spend_mid_usd": forecast["campaign_totals"]["daily_spend_mid_usd"],
        "unjoined_keywords": forecast["campaign_totals"]["unjoined_keywords"],
    }
    print(json.dumps(summary))
    return 0


if __name__ == "__main__":
    sys.exit(main_with_args(sys.argv))
```
